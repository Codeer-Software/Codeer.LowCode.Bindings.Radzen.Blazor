CREATE TABLE read_data_controls (
  id BIGSERIAL PRIMARY KEY,
  check_ BOOLEAN NULL,
  toggle BOOLEAN NULL,
  switch BOOLEAN NULL,
  date_ DATE NULL,
  date_time TIMESTAMP NULL,
  link TEXT NULL,
  number_ INTEGER NULL,
  radio TEXT NULL,
  select_ TEXT NULL,
  select_link TEXT NULL,
  text_ TEXT NULL,
  time_ TIME NULL,
  file_guid uuid NULL,
  file_name text NULL,
  file_size integer NULL
);

CREATE TABLE link_data (
  id BIGSERIAL PRIMARY KEY,
  code TEXT NULL,
  name TEXT NULL
);


CREATE TABLE temporary_files (
    id bigint GENERATED BY DEFAULT AS IDENTITY,
    guid uuid NOT NULL,
    created_date_time timestamp without time zone NULL,
    CONSTRAINT pk_temporary_files PRIMARY KEY (id)
);
